BEGIN DatabaseHelper

    DEFINE DATABASE_NAME = "inventory.db"
    DEFINE DATABASE_VERSION = 2

    // Inventory table columns
    DEFINE TABLE_ITEMS = "items"
    DEFINE COL_ID = "id"
    DEFINE COL_NAME = "name"
    DEFINE COL_QUANTITY = "quantity"
    DEFINE COL_PRICE = "price"

    // User table columns
    DEFINE TABLE_USERS = "users"
    DEFINE COL_USER_ID = "id"
    DEFINE COL_USERNAME = "username"
    DEFINE COL_PASSWORD = "password"
    DEFINE COL_ROLE = "role"

    CONSTRUCTOR(context):
        call parent SQLiteOpenHelper constructor with database name and version

    ON create(database):
        EXECUTE SQL:
            CREATE TABLE items (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                name TEXT,
                quantity INTEGER,
                price REAL
            )

        EXECUTE SQL:
            CREATE TABLE users (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                username TEXT UNIQUE,
                password TEXT,
                role TEXT
            )

    ON upgrade(database, oldVersion, newVersion):
        EXECUTE SQL: DROP TABLE IF EXISTS items
        EXECUTE SQL: DROP TABLE IF EXISTS users
        CALL onCreate(database)

    // -----------------------------
    // INVENTORY OPERATIONS
    // -----------------------------
    FUNCTION insertItem(name, quantity, price):
        OPEN writable database
        CREATE keyValueMap
            put name
            put quantity
            put price
        INSERT into items table using keyValueMap
        IF insert result is not -1:
            RETURN true
        ELSE:
            RETURN false

    FUNCTION getAllItems():
        OPEN writable database
        RUN SQL: SELECT * FROM items
        RETURN cursor

    // -----------------------------
    // USER ACCOUNT OPERATIONS
    // -----------------------------
    FUNCTION registerUser(username, password, role):
        OPEN writable database
        CREATE keyValueMap
            put username
            put password
            put role
        INSERT into users table using keyValueMap
        IF insert result is not -1:
            RETURN true
        ELSE:
            RETURN false

    FUNCTION verifyUser(username, password):
        OPEN writable database
        RUN SQL:
            SELECT * FROM users
            WHERE username = ? AND password = ?
            (bind username, password)
        RETURN cursor

END DatabaseHelper
